Date Validation PlayGround



$('input.type1').DateValidation({
    'beginYear': 1957,
    'endYear':2015,
    'chkExceedsToday':false,
    'attribute':'data-dv',
    'splitter':'.',
    'cssForValidDate':'pc-valid-date',
    'cssForInvalidDate':'pc-invalid-date'
});
